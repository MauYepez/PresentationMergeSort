// C++ program for Merge Sort

#include <iostream>
using namespace std;

                  
// Function to print the array.

void printArray(int* arr, int length) 
{
    for (int i = 0; i < length; i++) 
    {
        cout << arr[i] << " ";
    }
    cout << endl;
}


// Function to merge the sub-arrays.                  

void merge(int* arr, int left, int mid, int right) 
{
    int l, r, m;
    int const lengthLeft = mid - left + 1;
    int const lengthRight = right - mid;
  
    // Creating two temp arrays to store left and right sub-arrays.  
    int* arrLeft = new int[lengthLeft];
    int* arrRight = new int[lengthRight];
  
    // Copying the data from the actual array to left and right temp arrays.  
    for (int a = 0; a < lengthLeft; a++) 
    {
        arrLeft[a] = arr[left + a];
    }
    for (int a = 0; a < lengthRight; a++) 
    {
        arrRight[a] = arr[mid + 1 + a];
    }
  
    // Merging the temp arrays back into the actual array.  
    l = 0; // Starting index of arrLeft[].
    r = 0; // Staring index of arrRight[].
    m = left; // Starting index of merged subarray.
  
    while (l < lengthLeft && r < lengthRight) 
    {
        // Checking and placing the smaller number of both temp arrays into the main array.
        if (arrLeft[l] <= arrRight[r]) 
        {
            arr[m] = arrLeft[l];
            l++;
        }
        else 
        {
            arr[m] = arrRight[r];
            r++;
        }
        m++;
    }
  
    // After the successful execution of the above loop 
    // copying the remaining elements of the left subarray (if remaining). 
    while (l < lengthLeft) 
    {
        arr[m] = arrLeft[l];
        m++;
        l++;
    }

    // Copying the remaining elements of the right sub array (if remaining). 
    while (r < lengthRight) 
    {
        arr[m] = arrRight[r];
        m++;
        r++;
    }
}


                  
                  
// The mergeSort() function.                   

void mergeSort(int* arr, int left, int right) 
{
    int mid;

    if (left < right) 
    {
        // Calculating the mid.
        mid = (left + right) / 2;

        // Calling the function mergeSort() recursively and breaking down the given array into smaller sub arrays.
        mergeSort(arr, left, mid);
        mergeSort(arr, mid + 1, right);

        // Calling the merge() function to merge the sorted subarrays into the main array.
        merge(arr, left, mid, right);
    }
}

                  
                  


int main() 
{
    // Initialzing the array.
    int arr[] = { 9, 8, 15, 89, 1, 99, 5, 5, 20, 44 };
  
    // Calculating the length of the array.
    int length = sizeof(arr) / sizeof(int);
  
    // Printing the array before sorting.
    cout << "\n\tArray, before calling the mergeSort(): \n\t";
    printArray(arr, length);
  
    // Calling the mergeSort() function.
    mergeSort(arr, 0, length - 1);
  
    // Printing the array after sorting.
    cout << "\n\tArray, after calling the mergeSort(): \n\t";
    printArray(arr, length);
  
    return 0;
}

// C++ code to demonstrate the working of merge() 

#include <algorithm> 
#include <iostream>
#include <vector>

using namespace std;

 

int main()
{
    // initializing 1st container
    vector arr1 = { 9, 8, 15, 89, 1 };
    
    // initializing 2nd container  
    vector arr2 = { 99, 5, 5, 20, 44 };
    
    // declaring resultant container  
    vector finalArr(10);
    
    // sorting initial containers  
    sort(arr1.begin(), arr1.end());
    sort(arr2.begin(), arr2.end());
    
    // using merge() to merge the initial containers
    merge(arr1.begin(), arr1.end(), arr2.begin(), arr2.end(), finalArr.begin());
    
    // printing the resultant merged container
    cout << "\n\tThe container after merging initial containers is:\n\t";
    
    for (int i = 0; i < finalArr.size(); i++)
    {
        cout << finalArr[i] << " ";
    }
    cout << endl;
    
    return 0;
}
