// C++ program for Merge Sort

#include <iostream>
using namespace std;


void PrintArray(int* array, int length);
void DivideArray(int* array, int begin, int end);
void MergeSubarrays(int* array, int begin, int middle, int end);
    

int main() 
{
    // Initialzing the array.
    int main_array[] = { 9, 8, 15, 1, 99, 5, 5, 44 };
    
    // Calculating the length of the array.
    int length_main_array = sizeof(main_array) / sizeof(int);
    
    // Printing the array before sorting.
    cout << "\n\tArray, before using the MergeSort: \n\t";
    PrintArray(main_array, length_main_array);
    
    // Calling the mergeSort named DivideArray function.
    DivideArray(main_array, 0, length_main_array - 1);
    
    // Printing the array after sorting.
    cout << "\n\tArray, after using the MergeSort: \n\t";
    PrintArray(main_array, length_main_array);
    
    return 0;
}


// Function to print the array.

void PrintArray(int* main_array, int length_main_array)
{
    for (int iterator = 0; iterator < length_main_array; iterator++)
    {
        cout << main_array[iterator] << " ";
    }
    cout << endl;
}


                                  
// The mergeSort function named DivideArray() function.                   

void DivideArray(int* array, int begin, int end) 
{
    int middle;

    if (begin < end) 
    {
        // Calculating the middle.
        middle = (begin + end) / 2;

         // Calling the function DivideArray() recursively and breaking down the given array into smaller sub arrays.
        DivideArray(array, begin, middle);
        DivideArray(array, middle + 1, end);

        // Calling the MergeSubarrays() function to merge the sorted subarrays into the main array.
        MergeSubarrays(array, begin, middle, end);
    }
}

            

// Function to merge the sub-arrays.                  

void MergeSubarrays(int* main_array, int begin, int middle, int end)
{
    int index_A, index_B, index_main_array;
  
    int const length_A = middle - begin + 1;
    int const length_B = end - middle;
  
    // Creating two temp arrays to store left and right sub-arrays.    
    int* subarray_A = new int[length_A];
    int* subarray_B = new int[length_B];
  
    // Copying the data from the actual array to subarray_A and subarray_B temp arrays.  
    for (int iterator = 0; iterator < length_A; iterator++)
    {
        subarray_A[iterator] = main_array[begin + iterator];
    }
    for (int iterator = 0; iterator < length_B; iterator++)
    {
        subarray_B[iterator] = main_array[middle + 1 + iterator];
    }
  
    // Merging the temp arrays back into the actual array.  
    index_A = 0; // Starting index of subarray_A[].
    index_B = 0; // Staring index of subarray_B[].
    index_main_array = begin; // Starting index of main array.
  
    while (index_A < length_A && index_B < length_B) 
    {
        // Checking and placing the smaller number of both temp arrays into the main array.
        if (subarray_A[index_A] <= subarray_B[index_B])
        {
            main_array[index_main_array] = subarray_A[index_A];
            index_A++;
        }
        else
        {
            main_array[index_main_array] = subarray_B[index_B];
            index_B++;
        }
        index_main_array++;
    }
  
    // After the successful execution of the above loop 
    // copying the remaining elements of the subarray_A subarray (if remaining). 
    while (index_A < length_A)
    {
        main_array[index_main_array] = subarray_A[index_A];
        index_main_array++;
        index_A++;
    }

    // Copying the remaining elements of the subarray_B sub array (if remaining). 
    while (index_B < length_B)
    {
        main_array[index_main_array] = subarray_B[index_B];
        index_main_array++;
        index_B++;
    }
}

// C++ code to demonstrate the working of merge() 

#include <algorithm> 
#include <iostream>
#include <vector>

using namespace std;

 

int main()
{
    // initializing subarray_A
    vector<int> subarray_A = { 9, 8, 15, 1 };
    
    // initializing subarray_B  
    vector<int> subarray_B = { 99, 5, 5, 44 };
    
    // declaring resultant container  
    vector<int> main_array(8);
    
    // sorting initial containers  
    sort(subarray_A.begin(), subarray_A.end());
    sort(subarray_B.begin(), subarray_B.end());
    
    // using merge() to merge the initial containers
    merge(subarray_A.begin(), subarray_A.end(), subarray_B.begin(), subarray_B.end(), main_array.begin());
    
    // printing the resultant merged container
    cout << "\n\tsubarray_A:\n\t";
    for (int i = 0; i < subarray_A.size(); i++)
    {
        cout << subarray_A[i] << " ";
    }
    cout << endl;
    cout << "\n\tsubarray_B:\n\t";
    for (int i = 0; i < subarray_B.size(); i++)
    {
        cout << subarray_B[i] << " ";
    }
    cout << endl;
    cout << "\n\tThe container after merging initial containers is:\n\t";
    
    for (int i = 0; i < main_array.size(); i++)
    {
        cout << main_array[i] << " ";
    }
    cout << endl;
    
    return 0;
}
